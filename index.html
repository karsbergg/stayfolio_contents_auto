function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = JSON.parse(e.postData.contents);
    var hotels = Array.isArray(data) ? data : [data];

    if (sheet.getLastRow() === 0) {
      sheet.appendRow(["title","simple_desc","desc_1","desc_2",
        "special_title_1","special_desc_1",
        "special_title_2","special_desc_2",
        "special_title_3","special_desc_3",
        "status","timestamp"]);
    }

    hotels.forEach(function(h) {
      sheet.appendRow([
        h.title||"", h.simple_desc||"", h.desc_1||"", h.desc_2||"",
        h.special_title_1||"", h.special_desc_1||"",
        h.special_title_2||"", h.special_desc_2||"",
        h.special_title_3||"", h.special_desc_3||"",
        h.status||"ok", new Date().toISOString()
      ]);
    });

    return ContentService
      .createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false,error:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({ok:true,msg:"active"}))
    .setMimeType(ContentService.MimeType.JSON);
}
